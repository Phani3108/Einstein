Quick Start

# Authenticate
curl -X POST /api/v1/admin/login \
  -H "Content-Type: application/json" \
  -d '{"username": "...", "password": "..."}'

# Create a thought
curl -X POST /api/v1/thoughts \
  -H "Authorization: Bearer $TOKEN" \
  -d '{"content": "Had coffee with Sarah to discuss the project"}'

# Search your knowledge
curl /api/v1/search?q="meetings with Sarah" \
  -H "Authorization: Bearer $TOKEN"
